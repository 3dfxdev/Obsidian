Example:
    brushes =
    {
      {
        "solid",
        { x=0, y=50, face={ BLAH } },
        { x=80, y=50, face={ BLAH } },
        { x=40, y=90, face={ BLAH } },
        { b=100, face={ BLAH } },               -- bottom, -EXTREME_H if absent
        { t=160, slope={..}, face={ BLAH } },   -- top,    +EXTREME_H if absent
      },

      ...
    }
